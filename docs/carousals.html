<Carousel NumberOfItems="@items.Count" bind-ActiveIndex="@ActiveIndex" PauseOnHover="true">
  <CarouselIndicators NumberOfItems="@items.Count" bind-ActiveIndex="@ActiveIndex" />
  <div class="carousel-inner">
      @for (int i = 0; i < items.Count; i++)
      {
          <CarouselItem IsActive="@(ActiveIndex == i)" src="@items[i].Source" alt="@items[i].Alt" />
      }
  </div>
  <CarouselControl CarouselDirection="CarouselDirection.Previous" NumberOfItems="@items.Count" bind-ActiveIndex="@ActiveIndex" />
  <CarouselControl CarouselDirection="CarouselDirection.Next" NumberOfItems="@items.Count" bind-ActiveIndex="@ActiveIndex" />
</Carousel>
  
@functions {
  IList<Item> items = new List<Item>
  {
      new Item {
          Source = ".../First Slide",
          Alt ="First Slide",
          Caption = "Slide 1",
          Header = "Slide 1 Header"
      },
      new Item {
          Source = ".../Second Slide",
          Alt ="Second Slide",
          Caption = "Slide 2",
          Header = "Slide 2 Header"
      },
      new Item {
          Source = ".../Third Slide",
          Alt ="Third Slide",
          Caption = "Slide 3",
          Header = "Slide 3 Header"
      }
  };

  private int _activeIndex;
  int ActiveIndex
  {
      get => _activeIndex;
      set { _activeIndex = value; StateHasChanged(); }
  }

  class Item
  {
      public string Source { get; set; }
      public string Alt { get; set; }
      public string Caption { get; set; }
      public string Header { get; set; }
  }
}

<Carousel NumberOfItems="@items.Count" bind-ActiveIndex="@SecondIndex" PauseOnHover="true">
  <CarouselIndicators NumberOfItems="@items.Count" bind-ActiveIndex="@SecondIndex" />
  <div class="carousel-inner">
      @for (int i = 0; i < items.Count; i++)
      {
          Item item = items[i];
          <CarouselItem IsActive="@(SecondIndex == i)" src="@item.Source" alt="@item.Alt">
             <CarouselCaption CaptionText="@item.Caption" HeaderText="@item.Header" />
          </CarouselItem>
      }
  </div>
  <CarouselControl CarouselDirection="CarouselDirection.Previous" NumberOfItems="@items.Count" bind-ActiveIndex="@SecondIndex" />
  <CarouselControl CarouselDirection="CarouselDirection.Next" NumberOfItems="@items.Count" bind-ActiveIndex="@SecondIndex" />
</Carousel>
  
@functions {
  IList<Item> items = new List<Item>
  {
      new Item {
          Source = ".../First Slide",
          Alt ="First Slide",
          Caption = "Slide 1",
          Header = "Slide 1 Header"
      },
      new Item {
          Source = ".../Second Slide",
          Alt ="Second Slide",
          Caption = "Slide 2",
          Header = "Slide 2 Header"
      },
      new Item {
          Source = ".../Third Slide",
          Alt ="Third Slide",
          Caption = "Slide 3",
          Header = "Slide 3 Header"
      }
  };

  private int _secondIndex;
  int SecondIndex
  {
      get => _secondIndex;
      set { _secondIndex = value; StateHasChanged(); }
  }

  class Item
  {
      public string Source { get; set; }
      public string Alt { get; set; }
      public string Caption { get; set; }
      public string Header { get; set; }
  }
}
