Supplying OnClick Handler

<ListGroup ListGroupType="ListGroupType.Button">
    <ListGroupItem ListGroupType="ListGroupType.Button" OnClick="@onclick">Click me</ListGroupItem>
</ListGroup>

@functions {
    void onclick(UIMouseEventArgs e)
    {
        Console.WriteLine("Button was clicked!");
    }
}
