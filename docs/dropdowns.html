            }
        }

Split Button

Note that you can also make a DropdownItem into a seperator by adding IsDivider="true" to the DropdownItem.

Action Another Action
Yet Another Action 
        <ButtonGroup>
            <BlazorButton Color="Color.Primary">Action</BlazorButton>
            <DropdownToggle IsSplit="true" Color="Color.Primary" onclick="@onclick7" IsOpen="@IsOpen7" />
            <DropdownMenu IsOpen="@IsOpen7">        
                <DropdownItem Href="#">Action</DropdownItem>
                <DropdownItem Href="#">Another Action</DropdownItem>
                <DropdownItem IsDivider="true" />
                <DropdownItem Href="#">Yet Another Action</DropdownItem>
            </DropdownMenu>
        </ButtonGroup>

Menu Items

Menu items can be buttons

 
        <Dropdown IsOpen="@IsOpen16">
            <DropdownToggle Color="Color.Secondary" onclick="@onclick16" IsOpen="@IsOpen16">Dropdown</DropdownToggle>
            <DropdownMenu IsOpen="@IsOpen16">
                <DropdownItem IsButton="true" onclick="@menuButtonClick">Action</DropdownItem>
                <DropdownItem IsDivider="true" />
                <DropdownItem IsButton="true" onclick="@menuButtonClick">Yet Another Action</DropdownItem>
            </DropdownMenu>
        </Dropdown>

You can also create non-interactive dropdown items with .dropdown-item-text. Feel free to style further with custom CSS or text utilities.

            <span class="dropdown-item-text">Dropdown item text</span>
            <DropdownItem href="#">Action</DropdownItem>
            <DropdownItem href="#">Another action</DropdownItem>
            <DropdownItem href="#">Something else here</DropdownItem>
        </DropdownMenu>
