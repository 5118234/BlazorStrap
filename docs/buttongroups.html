Nesting

Dropdown link Dropdown link 
<ButtonGroup>
    <BlazorButton Color="Color.Secondary">1</BlazorButton>
    <BlazorButton Color="Color.Secondary">2</BlazorButton>

    <Dropdown IsGroup="true" IsOpen="@IsOpen">
        <DropdownToggle Color="Color.Secondary" onclick="@onclick" IsOpen="@IsOpen">Dropdown</DropdownToggle>
        <DropdownMenu IsOpen="@IsOpen">
            <DropdownItem href="#">Dropdown link</DropdownItem>
            <DropdownItem href="#">Dropdown link</DropdownItem>
        </DropdownMenu>
    </Dropdown>
</ButtonGroup>

@functions {
    bool IsOpen { get; set; }
    void onclick(UIMouseEventArgs e)
    {
        IsOpen = !IsOpen;
        StateHasChanged();
    }
}
